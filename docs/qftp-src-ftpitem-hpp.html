    #ifndef FTPITEM_HPP
    #define FTPITEM_HPP

    #include <QtCore/QDateTime>
    #include <QtCore/QVariant>

    /**
     * The FtpItem encapsulates the data for one item (file or directory) that
     * has been received from a FTP directory listing.
     */
    class FtpItem
    {
    public:
        /**
         * This operator is needed as a workaround for a bug in
         * QListDataModel::data() in Cascades RC10.
         */
        operator QVariant() const
        {
            return QVariant::fromValue(*this);
        }

        // The name of the item
        QString fileName;

        // The size of the item
        qint64 fileSize;

        // The owner of the item
        QString owner;

        // The group of the item
        QString group;

        // The creation time of the item
        QDateTime time;

        // Whether this item is a directory
        bool isDirectory;
    };

    Q_DECLARE_METATYPE(FtpItem)

    #endif
